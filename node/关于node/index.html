fs模块

fs: file systerm 文件系统 操作文件 读取/写入

1、使用此模块：let fs = require(‘fs’)

所有文件系统操作都具有同步和异步的形式

异步的形式会把回调函数作为最后一个参数；同步操作可以使用try...catch来处理异常

2、读取文件 （异步）

  • 无编码：解析数据需要将data转换为字符串格式
fs.readFile('./a.txt', (err, data) => {
    console.log(err);  //null 说明没有错误
    console.log(data);  //Buffer 是node的二进制流，可以转化为字符串
    console.log(data.toString());

    if (err) {
        console.log('文件读取失败');
        return
    } else {
        console.log(data.toString());
    }

    // 或者
    if (err) {
        return console.log('文件读取失败');
    }
    console.log(data.toString());
})
  • 有编码：在读取文件添加参数2：编码：‘utf8/utf-8’,效果等同于toString()
// 引入fs
let fs = require('fs');

// 2  读取
// 参数1：文件路径
// 参数2：编码 'utf-8/utf8'  ==> 等同于toString()
// 参数3：回调函数

fs.readFile('./a.txt', 'utf8', (err, data) => {
    if (err) {
        return console.log('读取文件失败');
    }
    console.log(data);
})
  • 读取文件的一个错误

ENOENT: no such file or directory

E : error(错误码都是E开头的) NO(没有) ENT(实体)

3、fs写入文件( fs.writeFile() )

// 1 引入fs
let fs = require('fs');

// 2 写入" 这是新写入的文件"
//  参数1：要写入的文件路径 (要写入到什么地方)
//  参数2：数据内容        (要写入的内容)
//  参数3：回调函数 (只有一个err的参数)
//        err  有值 => 写入失败  
//             没有值 =>  写入成功
//  如果文件不存在  将会创建一个新的文件
fs.writeFile('./a.txt', '这是新写入的文件', err => {
    if (err) {
        return console.log('写入失败');
    }
    console.log('写入成功');
    // 直接写入会把之前文件里面的数据直接覆盖掉
})

4、fs追加文件( fs.appendFile() )

// 1 引入
let fs = require('fs');

// 2、 追加
//  参数1：追加到哪里（文件路径）
//  参数2：追加什么内容
//  参数3：回调函数
fs.appendFile('./a.txt', '都洗洗睡吧', err => {
    if (err) {
        return console.log('追加失败')
    }

    console.log('追加成功')
})

//追加之后重新读取到最新的内容；
// 直接打印的data还是文件中原本的内容，追加的内容并没有读取到

__dirname 和 ./ 的区别

Node.js 中，__dirname 总是指向被执行 js 文件的绝对路径，所以当你在 /d1/d2/myscript.js 文件中写了 __dirname， 它的值就是 /d1/d2 。

相反，./ 会返回你执行 node 命令的路径，例如你的工作路径。

实现简单的http请求

// 1 引入
let fs = require('fs')
let http = require('http');
// 2 创建服务器
let server = http.createServer();
// 3 开启服务器
server.listen(8001, () => {
    console.log('开启成功了 http://localhost:8001');
    console.log('端口号localhost:' + server.address().port)
})
// 4 监听请求
server.on('request', (req, res) => {
    res.writeHead(200, {
        'content-type': 'text/html;charset:utf-8'
    })
    res.write(fs.readFileSync('./index.html'));
    // 4.1 url路径
    console.log(req.url);
    // 4.2 method 请求类型 GET/POST
    console.log(req.method);
    // 4.3 headers 请求头类型 {}
    console.log(req.headers);

    console.log('有人请求了');
    res.end()
})

res.end()

  • 结束响应
  • res.end()这个方法告诉服务器所有要发送的响应头和响应体都发送完毕。可以人为结束这次响应
  • 同时每次响应都必须调用该方法，用来结束响应
  • 参数1：结束响应前要发送的数据，选填
  • 参数2：编码，选填
  • 参数3：回调函数，选填

request请求事件处理函数

request请求事件处理函数，需要接收两个参数

  • request 请求对象

    请求对象那可以用来获取客户端发送的一些请求信息

    res.url : 请求路径

    res.methods : 请求方式

    res.headers : 请求头

  • response 响应对象

    res.write :

    • 参数1：要写入的数据，可以是字符串或二进制，必填

    • 参数2：编码，默认是‘utf8’ ,选填

    • 参数3：回调函数，选填

    • write 可以使用多次，但是最后一次一定要使用end来结束响应，否则可会断会一直等待

    res.end()：告诉客户端，响应头和响应体都已经发送完毕，可以呈递给用户了

    res.setHeader(name,value):设置响应报文头

    res.statusCode:设置或读取http响应码

    res.statusMessage:设置或读取http响应状态信息

    // 1、引入
let http = require('http');
// 2、创建服务器
let http1 = http.createServer();
// 3、开启服务器
http1.listen('8001', () => {
    console.log('成功开启服务器 http://localhost:8001');
})
// 4、监听服务器
http1.on('request', (req, res) => {
    // 4.1、请求路径
    console.log(req.url);
    // 4.2 请求头
    console.log(req.headers);
    // 4.3 请求方式
    console.log(req.methods);
  
    // 告诉浏览器解析内容 也要按照utf8 出现汉字需要设置
    res.setHeader('content-type', 'text/plain;charset=utf8')
    res.write('哈哈哈哈哈');  //写入页面中，如果是中文需要设置响应头
  
    // 设置或读取http响应码
    console.log(res.statusCode);  //200
    res.statusCode = 202
    console.log('22', res.statusCode);  //202
    // 设置或读取http响应状态信息
    res.statusMessage = 'yes'
    console.log(res.statusMessage);   //yes
  
    res.end()
})

node.js中的重定向

重定向的准则

  1. 给客户端发送301响应代码，告诉客户：资源已经移动到另外一个位置
  2. 发送一个位置头(Lacation Header) 告诉客户重定向到哪里
var http = require('http');

http.createServer(function (req, res) {
  res.writeHead(301, {
    'Location': 'http://www.homestarrunner.com/sbsite/'
  });
  res.end();
}).listen(3000, "127.0.0.1");
console.log('Server running at http://127.0.0.1:3000/');

通过服务器让客户端重定向

  1. 状态码设置为 302 临时重定向 statusCode
  2. 在响应头中通过 Location 告诉客户端往哪儿重定向 setHeader 响应头内容 如果客户端发现收到服务器的响应的状态码是 302 就会自动去响应头中找 Location ，然后对该地址发起新的请求 所以你就能看到客户端自动跳转了
  res.statusCode = 302
  res.setHeader('Location', '/')  //跳转到首页

Node 模块化

模块化规范分类：

  • 服务器端(node.js) (es5)
    • commonJS: require exports
  • ES6标准模块化规范(vue)
    • import export

在node.js中，应用由模块组成，node.js采用commonJS模块规范

  1. 一个js文件就是一个模块，每个js文件都独立运行在一个闭包中，不是全局的作用域，模块与模块无法互相访问
  2. 每个模块都是独立的作用域，在这个文件中定义的变量、函数、对象都是私有的，对其他文件不可见

Node 模块的三大类

  1. 内置模块（内置模块是由Node.js 官方提供的，例如fs、 http、 path等）
    1. 直接引入
    2. 直接使用
  2. 自定义模块（用户创建的每个js文件，都是自定义模块）
    1. 创建自定义模块
    2. 有引入（相对路径）(require)
    3. 有导出 (exports)
    4. 使用
  3. 第三方模块（eg: npm i mime）
    1. 需要安装/下载 ( npm cnpm yarn )
    2. 直接引入
    3. 直接使用

自定义模块

导出：每个js文件里面默认都有一个模块变量 module

​ 每个js文件默认导出的是module里面的exports

引入：通过require引入的模块变量

​ 引入的永远是module.exports的值，可以是任意类型(对象，数组，字符串，数字…)

npm的模块化

exports 和 module.exports

  1. exports 和 module.exports 指向同一块空间
  2. exports 是module.exports的一个别名
  3. 导出的永远是module.exports 的值
  4. exports只能使用.语法来向外暴露内部变量 exports.xxx = xxx; Module.exports既可以通过.语法，也可以直接赋值一个对象
  5. module.exports 初始值为一个空对象 {}
  6. exports 是指向的 module.exports 的引用
  • 注意： 如果用exports来导出对象的话，就会赋值给一个新的对象，那么和module.exports就没有关系了，因为在node里相当于做了let exports = module.exports;
//b.js
module.exports = {
    age: 20
}
exports = {
    name: 'oo'
}
//a.js
let res = require('./b.js')
console.log(res);   //{age:20}

module.exports 和 exports 一起使用

//b.js
module.exports.age = 20
exports.name = 'oo'
//a.js
let res = require('./b.js')
console.log(res);   //{ age: 20, name: 'oo' }

//第二种
module.exports.age = 20
exports.eat = function(){}
//a.js
let res = require('./b.js')
console.log(res);   //{ age: 20, eat: [function] }

Express

注册路由的三种方式

  1. app.METHOD ==> app.get/ post /put 等等之类 ⭐⭐⭐
    • 类型：完全匹配
    • 路径：完全匹配(path) /index 可以匹配到 /index/123 不可以匹配 /index?id=2 可以匹配
  2. app.all ⭐
    • 类型：任意类型
    • 路径：完全匹配(path)
  3. app.use ⭐⭐⭐
    • 类型：任意类型
    • 路径：以path开头的，类似于startsWith(); index/123 可以匹配； /index/list/123 可以匹配； /index123 不可以匹配
    • 如果路径为’/',则可以直接省略

express处理静态资源

使用 Express 中的 express.static 内置中间件函数，通过如下代码就可以将 public 目录下的图片、CSS 文件、JavaScript 文件对外开放访问了：

app.use(express.static('public'))

如果要使用多个静态资源目录，请多次调用 express.static 中间件函数。

为express提供服务的文件创建虚拟路径前缀(其中路径在文件系统中并不实际存在)。指定静态目录的挂载路径，如下所示:

app.use('/static', express.static('public'))

现在可以通过带有 /static 前缀地址来访问 public 目录中的文件了。

推荐使用绝对路径来加载静态资源

app.use('/static', express.static(path.join(__dirname,'./public')))
